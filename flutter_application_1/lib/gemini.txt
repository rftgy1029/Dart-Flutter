import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return const MaterialApp(
      home: CounterScreen(), // 우리가 만들 화면을 불러옵니다.
    );
  }
}

// 1. 상태(데이터)가 변하는 화면을 만들 때는 StatefulWidget을 사용합니다.
class CounterScreen extends StatefulWidget {
  const CounterScreen({super.key});

  @override
  State<CounterScreen> createState() => _CounterScreenState();
}

class _CounterScreenState extends State<CounterScreen> {
  // 2. 화면에서 관리할 변수(상태)를 선언합니다.
  int _counter = 0;

  // 3. 버튼이 눌렸을 때 실행할 함수를 만듭니다.
  void _incrementCounter() {
    setState(() {
      // setState는 "데이터가 바뀌었으니 화면을 다시 그려라!"라고 플러터에게 알리는 함수입니다.
      _counter++; 
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('플러터 카운터 앱'),
        backgroundColor: Colors.teal,
      ),
      // 4. 위젯들을 세로로 정렬하기 위해 Column 위젯을 사용합니다.
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            const Text(
              '버튼을 누른 횟수:',
              style: TextStyle(fontSize: 18.0),
            ),
            Text(
              '$_counter', // 변수 앞에는 $를 붙여 화면에 표시합니다.
              style: const TextStyle(fontSize: 48.0, fontWeight: FontWeight.bold),
            ),
          ],
        ),
      ),
      // 5. 화면 우측 하단에 떠 있는 버튼(FloatingActionButton)을 배치합니다.
      floatingActionButton: FloatingActionButton(
        onPressed: _incrementCounter, // 버튼을 누르면 위에서 만든 함수가 실행됩니다.
        backgroundColor: Colors.teal,
        child: const Icon(Icons.add), // 플러스 모양 아이콘
      ),
    );
  }
}
